SchemaBuilder UnifiedAutomation.UaServer.BaseNodeManager.CreateTypeDictionary ( RequestContext  context,
CreateTypeDictionarySettings  settings 
)

Creates a class that can be used to dynamically build data types descriptions.

Parameters
contextThe request context.
settingsThe settings used to initialize the dictionary.
Returns
The new schema dictionary object.

Each object creates an OPC Binary and XML Schema dictionary for the types added to it.

Each object has a NamespaceURI which is used for both dictionaries. By default the same URI is used for the NodeIds created to represent the types, however, this can be overridden.

When creating dictionaries the number of types should not get too large. The simplest way to control this is to create a multiple dictionaries with different URIs.

Example for creating DataTypeDictionaries with a structured DataType and an enumerated DataType and Variable nodes having values of the structured DataType:

static class GenericStructureIdentifiers
{
// Identifiers for DataType and Encoding nodes. Using numeric NodeIds in this example.
// For some use cases it is required to use string NodeIds. This is also possible but less
// efficient.
public const uint PersonDataTypeId = 543210;
public const uint PersonBinaryEncodingId = 543211;
public const uint PersonXmlEncodingId = 543212;
public const uint GenderDataTypeId = 543220;
// The NamespaceUri for the schema should be unique. Best practice is using the NamespaceUri of
// NodeManager for the first dictionary. If the NodeManager contains a dictionary already
// (e.g. the dictionary is created using UaModeler), you must NOT use the same NamespaceUri again.
public const string NamespaceUriForSchema = "http://www.unifiedautomation.com/Demo2/";
}
void SetupGenericDataTypeNodes()
{
var settings = new CreateTypeDictionarySettings()
{
NamespaceUriForSchema = GenericStructureIdentifiers.NamespaceUriForSchema,
BrowseName = new QualifiedName("Demo2", DefaultNamespaceIndex),
BinaryDictionaryNodeId = new NodeId("OPCBinary.Demo2", DefaultNamespaceIndex),
XmlDictionaryNodeId = new NodeId("XMLSchema.Demo2", DefaultNamespaceIndex)
};
// create a schema builder to manage the OPC binary and XML schema dictionaries.
var builder = CreateTypeDictionary(Server.DefaultRequestContext, settings);
NewEnumerationSettings enumerationSettings = new NewEnumerationSettings()
{
Name = "Gender",
DataTypeIdentifier = GenericStructureIdentifiers.GenderDataTypeId
};
GenericEnumeratedDataType enumerationType = builder.NewEnumeration(enumerationSettings);
enumerationType.Add("Male", 0);
enumerationType.Add("Female", 1);
enumerationType.Add("Diverse", 2);
CreateGenericDataType(Server.DefaultRequestContext, enumerationType, DefaultNamespaceIndex);
NewStructureSettings structureSettings = new NewStructureSettings()
{
Name = "Person",
DataTypeIdentifier = GenericStructureIdentifiers.PersonDataTypeId,
BinaryEncodingIdentifier = GenericStructureIdentifiers.PersonBinaryEncodingId,
XmlEncodingIdentifier = GenericStructureIdentifiers.PersonXmlEncodingId
};
GenericStructureDataType structureType = builder.NewStructure(structureSettings);
builder.AddField(structureType, "Name", BuiltInType.String);
builder.AddField(structureType, "Height", BuiltInType.UInt16);
builder.AddField(structureType, "Weight", BuiltInType.Float);
builder.AddField(structureType, "Gender", enumerationType);
CreateGenericDataType(Server.DefaultRequestContext, structureType, DefaultNamespaceIndex);
}

bool UnifiedAutomation.UaServer.BaseNodeManager.DisableAttributeReadAccessCheck

Gets or sets a value indicating whether the access right check for ReadAttribute is disabled for in memory nodes.

true if access right check for ReadAttribute is disabled; otherwise, false.

UserAccessMask.ReadAttribute has been added in SDK version 2.5. This enum value allows to control access to all attributes but the Value attribute. In older versions of the SDK all non-value attributes could be read. This property can be used to reset to old behavior.
